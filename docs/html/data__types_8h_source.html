157  }
158  case type_id::D_FLOAT: {
159  size = sizeof(float);
160  break;
161  }
162  case type_id::D_DOUBLE: {
163  size = sizeof(double);
164  break;
165  }
166  case type_id::D_STRING: {
167  break;
168  }
169  default: {

171  "Must specify length for non-primitive data types");
172  }
173  }
