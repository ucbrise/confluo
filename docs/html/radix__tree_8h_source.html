304  node_t* child = nullptr;
305  if ((child = atomic::load(&(node->children()[key[d]]))) == nullptr) {
306  // Try & allocate child node
307  child = new node_t(key[d], width_, d + 1, node);
308  node_t* expected = nullptr;
309 
310  // If thread was not successful in swapping newly allocated memory,
311  // then it should de-allocate memory, and accept whatever the
312  // successful thread allocated as the de-facto storage for child node.
313  if (!atomic::strong::cas(&(node->children()[key[d]]), &expected,
314  child)) {
315  delete child;
316  child = expected;
317  }

321  }
