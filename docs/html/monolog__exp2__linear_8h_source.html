313  __atomic_bucket_ref* try_allocate_container(size_t container_idx) {
314  size_t num_buckets = 1U << (container_idx + FCB_HIBIT);
315  __atomic_bucket_ref* new_container = new __atomic_bucket_ref[num_buckets]();
316  __atomic_bucket_ref* expected = nullptr;
317 
318  // Only one thread will be successful in replacing the nullptr reference with newly

320  if (!atomic::strong::cas(&bucket_containers_[container_idx], &expected, new_container)) {
321  // All other threads will deallocate the newly allocated container.
322  delete[] new_container;
323  return expected;
324  }

326  }
