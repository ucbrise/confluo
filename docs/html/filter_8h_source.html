117  void update(const record_t& r) {

119  aggregated_reflog* refs = idx_.insert(
120  byte_string(r.timestamp() / configuration_params::TIME_RESOLUTION_NS),

122  int tid = thread_manager::get_id();

142  void update(size_t log_offset, const schema_snapshot& snap,
143  record_block& block, size_t record_size) {
144  int tid = thread_manager::get_id();
145  aggregated_reflog* refs = nullptr;
146  std::vector<numeric> local_aggs;
147 
148  for (size_t i = 0; i < block.nrecords; i++) {
149  void* cur_rec = reinterpret_cast<uint8_t*>(&block.data[i * record_size]);
150  uint64_t rec_off = log_offset + i * record_size;
151  if (exp_.test(snap, cur_rec)) {
152  if (refs == nullptr) {
153  refs = idx_.get_or_create(
154  byte_string(static_cast<uint64_t>(block.time_block)), triggers_);
155  local_aggs.resize(refs->num_aggregates());
156  }
157  refs->push_back(rec_off);
158  for (size_t j = 0; j < local_aggs.size(); j++)
159  if (triggers_.at(j)->is_valid())
160  local_aggs[j] = triggers_.at(j)->agg(local_aggs[j], snap, cur_rec);
161  }
162  }
163 
164  size_t version = log_offset + block.nrecords * record_size;
165  for (size_t j = 0; j < local_aggs.size(); j++)
166  if (triggers_.at(j)->is_valid()
167  && local_aggs[j].type().id != type_id::D_NONE)
168  refs->update_aggregate(tid, j, local_aggs[j], version);
169  }
