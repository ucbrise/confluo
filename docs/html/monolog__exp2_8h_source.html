306  protected:
307  /* Tries to allocate the specifies bucket. If another thread has
308  * already succeeded in allocating the bucket, the current thread
309  * deallocates and returns.
310  * @param bucket_idx The index of the specified bucket
311  * @return The specified bucket
312  */
313  T* try_allocate_bucket(size_t bucket_idx) {
314  size_t size = (1U << (bucket_idx + FBS_HIBIT));
315  T* new_bucket = new T[size];
316  T* expected = NULL;
317 
318  // Only one thread will be successful in replacing the NULL reference with newly
319  // allocated bucket.
320  if (!atomic::strong::cas(&buckets_[bucket_idx], &expected, new_bucket)) {
321  // All other threads will deallocate the newly allocated bucket.
322  delete[] new_bucket;
323  return expected;
324  }
325 
326  return new_bucket;
327  }
