87  configuration_params::INDEX_BUCKET_SIZE) {
88  optional<management_exception> ex;
89  auto ret =
90  mgmt_pool_.submit(
91  [field_name, bucket_size, &ex, this] {
92 
93  uint16_t idx;
94  try {
95  idx = schema_.get_field_index(field_name);
96  } catch (std::exception& e) {
97  ex = management_exception("Could not add index for " + field_name + " : " + e.what());
98  return;
99  }
100 
101  column_t& col = schema_[idx];
102  bool success = col.set_indexing();
103  if (success) {
104  uint16_t index_id = UINT16_MAX;
105  switch (col.type().id) {
106  case type_id::D_BOOL:
107  index_id = indexes_.push_back(new radix_index(1, 2));
108  break;
109  case type_id::D_CHAR:
110  case type_id::D_SHORT:
111  case type_id::D_INT:
112  case type_id::D_LONG:
113  case type_id::D_FLOAT:
114  case type_id::D_DOUBLE:
115  case type_id::D_STRING:
116  index_id = indexes_.push_back(new radix_index(col.type().size, 256));
117  break;
118  default:
119  col.set_unindexed();
120  ex = management_exception("Index not supported for field type");
121  return;
122  }
123  col.set_indexed(index_id, bucket_size);
124  metadata_.write_index_info(field_name, bucket_size);
125  } else {
126  ex = management_exception("Could not index " + field_name + ": already indexed/indexing");
127  return;
128  }
129  });
130 
131  ret.wait();
132  if (ex.has_value())
133  throw ex.value();

259  size_t append_batch(record_batch& batch) {
260  size_t record_size = schema_.record_size();
261  size_t batch_bytes = batch.nrecords * record_size;
262  size_t log_offset = data_log_.reserve(batch_bytes);
263  size_t cur_offset = log_offset;
264  for (record_block& block : batch.blocks) {
265  data_log_.write(cur_offset,
266  reinterpret_cast<const uint8_t*>(block.data.data()),
267  block.data.length());
268  update_aux_record_block(cur_offset, block, record_size);
269  cur_offset += block.data.length();
270  }
271 
272  data_log_.flush(log_offset, batch_bytes);
273  rt_.advance(log_offset, batch_bytes);
274  return log_offset;
275  }

369  size_t record_size) {
370  schema_snapshot snap = schema_.snapshot();
371  for (size_t i = 0; i < filters_.size(); i++) {
372  if (filters_.at(i)->is_valid()) {
373  filters_.at(i)->update(log_offset, snap, block, record_size);

376 
377  for (size_t i = 0; i < schema_.size(); i++) {
378  if (snap.is_indexed(i)) {
379  radix_index* idx = indexes_.at(snap.index_id(i));
380  // Handle timestamp differently

384  size_t idx = refs->reserve(block.nrecords);
385  for (size_t j = 0; j < block.nrecords; j++) {
386  refs->set(idx + j, log_offset + j * record_size);
387  }
388  } else {

401  void monitor_task() {
402  uint64_t cur_ms = time_utils::cur_ms();
403  uint64_t version = rt_.get();
404  size_t nfilters = filters_.size();
405  for (size_t i = 0; i < nfilters; i++) {
406  filter* f = filters_.at(i);
407  if (f->is_valid()) {
408  for (uint64_t ms = cur_ms - configuration_params::MONITOR_WINDOW_MS;
409  ms <= cur_ms; ms++) {
410  const aggregated_reflog* ar = f->lookup(ms);
411  size_t ntriggers = f->num_triggers();
412  for (size_t tid = 0; tid < ntriggers; tid++) {
413  trigger* t = f->get_trigger(tid);
414  if (t->is_valid() && ar != nullptr) {
415  numeric agg = ar->get_aggregate(tid, version);
416  if (numeric::relop(t->op(), agg, t->threshold())) {
417  alerts_.add_alert(ms, t->trigger_name(), t->trigger_expr(), agg,
418  version);
419  }
420  }
421  }
422  }
423  }
