249  const expression_lex_token next_token() {
250  while (iswspace(stream_.peek()))
251  stream_.get();
252 
253  char c = stream_.get();
254  switch (c) {
255  case EOF:
256  return expression_lex_token(END, "");
257  case '|': {
258  if (stream_.get() != '|')
259  THROW(parse_exception,
260  "Invalid token starting with |; did you mean ||?");
261  return expression_lex_token(OR, "||");
262  }
263  case '&': {
264  if (stream_.get() != '&')
265  THROW(parse_exception,
266  "Invalid token starting with &; did you mean &&?");
267  return expression_lex_token(AND, "&&");
268  }
269  case '!': {
270  char c1 = stream_.get();
271  if (c1 == '=')
272  return expression_lex_token(OPERATOR, "!=");
273  if (c1 == 'i') {
274  if (stream_.get() == 'n' && iswspace(stream_.peek()))
275  return expression_lex_token(OPERATOR, "!in");
276  stream_.unget();
277  }
278  stream_.unget();
279  return expression_lex_token(NOT, "!");
280  }
281  case '(':
282  return expression_lex_token(LEFT, "(");
283  case ')':
284  return expression_lex_token(RIGHT, ")");
285  case '=': {
286  if (stream_.get() != '=')
287  THROW(parse_exception,
288  "Invalid token starting with =; did you mean ==?");
289  return expression_lex_token(OPERATOR, "==");
290  }
291  case '<': {
292  if (stream_.get() == '=')
293  return expression_lex_token(OPERATOR, "<=");
294  stream_.unget();
295  return expression_lex_token(OPERATOR, "<");
296  }
297  case '>': {
298  if (stream_.get() == '=')
299  return expression_lex_token(OPERATOR, ">=");
300  stream_.unget();
301  return expression_lex_token(OPERATOR, ">");
302  }
303  default: {
304  if (!opvalid(c))
305  THROW(parse_exception, "All operands must conform to [a-zA-Z0-9_.]+");
306 
307  stream_.unget();
308  std::string operand = "";
309  while (opvalid(stream_.peek()))
310  operand += (char) stream_.get();
311  return expression_lex_token(OPERAND, operand);
312  }
313  }
314  return expression_lex_token(INVALID, "");
315  }

370  expression_t* factor() {
371  expression_lex_token tok = lex_.next_token();
372  if (tok.id == expression_lexer::NOT) {
373  return negate(factor());
374  } else if (tok.id == expression_lexer::LEFT) {
375  expression_t *e = exp();
376  expression_lex_token right = lex_.next_token();
377  if (right.id != expression_lexer::RIGHT)
378  THROW(parse_exception, "Could not find matching right parenthesis");
379  return e;
380  } else if (tok.id == expression_lexer::OPERAND) {
381  predicate_t *p = new predicate_t;
382  p->attr = tok.value;
383  expression_lex_token op = lex_.next_token();
384  if (op.id != expression_lexer::OPERATOR)
385  THROW(parse_exception,
386  "First operand must be followed by operator in all predicates");
387  p->op = relop_utils::str_to_op(op.value);
388  expression_lex_token operand = lex_.next_token();
389  if (operand.id != expression_lexer::OPERAND)
390  THROW(parse_exception,
391  "Operand must be followed by operator in all predicates");
392  p->value = operand.value;
393  return p;
394  } else {
395  THROW(parse_exception, "Unexpected token " + tok.value);
396  }
