153 static struct elias_gamma_prefix_sum {
154  public:
155  typedef uint16_t block_type;
156 
157  elias_gamma_prefix_sum() {
158  for (uint64_t i = 0; i < 65536; i++) {
159  uint16_t val = (uint16_t) i;
160  uint64_t count = 0, offset = 0, sum = 0;
161  while (val && offset <= 16) {
162  int N = 0;
163  while (!GETBIT(val, offset)) {
164  N++;
165  offset++;
166  }
167  offset++;
168  if (offset + N <= 16) {
169  sum += ((val >> offset) & ((uint16_t) low_bits_set[N])) + (1 << N);
170  offset += N;
171  count++;
172  } else {
173  offset -= (N + 1);
174  break;
175  }
176  }
177  prefixsum_[i] = (offset << 24) | (count << 16) | sum;
178  }
179  }
180 
181  uint8_t offset(const block_type i) const {
182  return ((prefixsum_[(i)] >> 24) & 0xFF);
183  }
184 
185  uint8_t count(const block_type i) const {
186  return ((prefixsum_[i] >> 16) & 0xFF);
187  }
188 
189  uint16_t sum(const block_type i) const {
190  return (prefixsum_[i] & 0xFFFF);
191  }

193  private:
194  uint32_t prefixsum_[65536];

255  T prefix_sum(pos_type delta_offset, pos_type until_idx) {
256  T delta_sum = 0;
257  pos_type delta_idx = 0;
258  pos_type current_delta_offset = delta_offset;
259  while (delta_idx != until_idx) {
260  uint16_t block = this->deltas_->template get_val_pos<uint16_t>(
261  current_delta_offset, 16);
262  uint16_t cnt = elias_gamma_prefix_table.count(block);
263  if (cnt == 0) {
264  // If the prefixsum table for the block returns count == 0
265  // this must mean the value spans more than 16 bits
266  // read this manually
267  width_type delta_width = 0;
268  while (!this->deltas_->get_bit(current_delta_offset)) {
269  delta_width++;
270  current_delta_offset++;
271  }
272  current_delta_offset++;
273  delta_sum += this->deltas_->template get_val_pos<T>(
274  current_delta_offset, delta_width) + (1ULL << delta_width);
275  current_delta_offset += delta_width;
276  delta_idx += 1;
277  } else if (delta_idx + cnt <= until_idx) {
278  // If sum can be computed from the prefixsum table
279  delta_sum += elias_gamma_prefix_table.sum(block);
280  current_delta_offset += elias_gamma_prefix_table.offset(block);
281  delta_idx += cnt;
282  } else {
283  // Last few values, decode them without looking up table
284  while (delta_idx != until_idx) {
285  width_type delta_width = 0;
286  while (!this->deltas_->get_bit(current_delta_offset)) {
287  delta_width++;
288  current_delta_offset++;
289  }
290  current_delta_offset++;
291  delta_sum += this->deltas_->template get_val_pos<T>(
292  current_delta_offset, delta_width) + (1ULL << delta_width);
293  current_delta_offset += delta_width;
294  delta_idx += 1;
295  }
