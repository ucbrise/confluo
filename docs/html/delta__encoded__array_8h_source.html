186  elias_gamma_prefix_sum() {
187  for (uint64_t i = 0; i < 65536; i++) {
188  uint16_t val = (uint16_t) i;
189  uint64_t count = 0, offset = 0, sum = 0;
190  while (val && offset <= 16) {
191  int N = 0;
192  while (!GETBIT(val, offset)) {
193  N++;
194  offset++;
195  }
196  offset++;
197  if (offset + N <= 16) {
198  sum += ((val >> offset) & ((uint16_t) low_bits_set[N])) + (1 << N);
199  offset += N;
200  count++;
201  } else {
202  offset -= (N + 1);
203  break;
204  }
205  }
206  prefixsum_[i] = (offset << 24) | (count << 16) | sum;
207  }
208  }
