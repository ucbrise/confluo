90  bool find(const K& key, V& val, marked_ptr_t** oprev = nullptr,
91  marked_ptr_t* ocur = nullptr, marked_ptr_t* onext = nullptr) {
92  K ckey;
93  marked_ptr_t *prev = nullptr;
94  marked_ptr_t cur, next;
95 
96  while (true) {
97  prev = head_;
98  cur = *prev;
99  while (true) {
100  if (cur.ptr() == nullptr) {
101  if (oprev)
102  *oprev = prev;
103  if (ocur)
104  *ocur = cur;
105  if (onext)
106  *onext = next;
107  return false;
108  }
109 
110  next = cur->next;
111  ckey = cur->key;
112  val = cur->value;
113 
114  // this means someone mucked with the list; start over
115  if (*prev != cur.unmarked())
116  break;
117 
118  // if next pointer is not marked
119  if (!next.marked()) {
120  // if current key > key, the key isn't in the list;
121  // if current key == key, the key IS in the list
122  if (ckey >= key) {
123  if (oprev)
124  *oprev = prev;
125  if (ocur)
126  *ocur = cur;
127  if (onext)
128  *onext = next;
129  return ckey == key;
130  }
131  } else {
132  marked_ptr_t expected = cur.unmarked();
133  if (atomic::c11::strong::cas(prev, &expected, next.unmarked())) {
134  delete cur.ptr();
135  } else {
136  break;
137  }
138  }
139  cur = next;
140  }
141  }
