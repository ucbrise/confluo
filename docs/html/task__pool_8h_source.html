47  bool dequeue(function_t& out) {
48  std::unique_lock<std::mutex> lock(mutex_);
49  condition_.wait(lock, [this]() {return !queue_.empty() || !valid_;});
50 
51  /*
52  * Using the condition in the predicate ensures that spurious wakeups with a valid
53  * but empty queue will not proceed, so only need to check for validity before proceeding.
54  */
55  if (!valid_)
56  return false;
57 
58  out = std::move(queue_.front());
59  queue_.pop();
60  return true;
61  }
