58  bool dequeue(function_t& out) {
59  std::unique_lock<std::mutex> lock(mutex_);
60  condition_.wait(lock, [this]() {return !queue_.empty() || !valid_;});

64  * but empty queue will not proceed, so only need to check for validity before proceeding.
